488  void TAngle<T>::normalize(Range range)
489  {
490  if (range == RANGE__UNLIMITED)
491  {
492  return;
493  }
494 
495  long mod_factor = (long)(value / (2 * Constants::PI));
496  value -= mod_factor * (Constants::PI * 2);
497 
498  while (Maths::isGreater(value, (Constants::PI * 2)))
499  {
500  value -= (Constants::PI * 2);
501  }
502  while (Maths::isLess(value, -(Constants::PI * 2)))
503  {
504  value += (Constants::PI * 2);
505  }
506  if (range == RANGE__SIGNED) // invariant: -180 to 180:
507  {
508  if (Maths::isGreater(value, Constants::PI))
509  {
510  value -= (Constants::PI * 2);
511  }
512  }
513  else
514  { // invariant: 0 to 360:
515  if (Maths::isLess(value, 0))
516  {
517  value += (Constants::PI * 2);
518  }
