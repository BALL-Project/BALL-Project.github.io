220  number_of_cells_y = (Size) from.getSize().y - 1;
221 
222  for (act_cell_y = 0; act_cell_y < number_of_cells_y; act_cell_y++)
223  {
224  for (act_cell_x = 0; act_cell_x < number_of_cells_x; act_cell_x++)
225  {
226  // First we have to find out the topology of the actual square.
227  int topology = 0;
228

232  }
