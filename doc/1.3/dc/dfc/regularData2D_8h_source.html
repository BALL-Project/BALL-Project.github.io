541  template <typename ValueType>
542  void TRegularData2D<ValueType>::rescale(const typename TRegularData2D<ValueType>::IndexType& size)
543  throw(Exception::OutOfMemory)
544  {
545  // If the old size equals the new size, we're done.
546  if ((size.x == size_.x) && (size_.y == size.y))
547  {
548  return;
549  }
550 
551  // If the new grid is empty, this whole thing is quite easy.
552  if ((size.x == 0) || (size.y == 0))
553  {
554  data_.resize(0);
555  dimension_.set(0.0);
556  return;
557  }
558 
559  // Compute the new array size.
560  size_type new_size = (size_type)(size.x * size.y);
561 
562  // Catch any bad_allocs thrown by vector::resize
563  try
564  {
565  // Create a new temporary array.
566  TRegularData2D<ValueType> old_data(*this);
567 
568  // Resize the data to its new size.
569  data_.resize(new_size);
570  spacing_.x = dimension_.x / (double)(size.x - 1);
571  spacing_.y = dimension_.y / (double)(size.y - 1);
572 
573  // Walk over the new grid and copy the (interpolated) old stuff back.
574  CoordinateType v;
575  for (size_type i = 0; i < new_size; i++)
576  {
577  Position x = i % size.x;
578  Position y = i / size.x;
579  v.x = origin_.x + x * spacing_.x;
580  v.y = origin_.y + y * spacing_.y;
581  data_[i] = old_data(v);
582  }
583 
584  // Correct the grid dimension. Origin and spacing remain constant.
585  size_ = size;
586  }
587  catch (std::bad_alloc&)
588  {
589  throw Exception::OutOfMemory(__FILE__, __LINE__, new_size * (Size)sizeof(ValueType));
590  }
591  }

594  void TRegularData2D<ValueType>::resize(const typename TRegularData2D<ValueType>::IndexType& size)
595  throw(Exception::OutOfMemory)
596  {
597  // If the old size equals the new size, we're done.
598  if (size.x == size_.x && size_.y == size.y)
599  {
600  return;
601  }
602 
603  // If the new grid is empty, this whole thing is quite easy.
604  if ((size.x == 0) || (size.y == 0))
605  {
606  data_.resize(0);
607  dimension_.set(0.0, 0.0);
608  return;
609  }
610 
611  // Compute the new array size.
612  size_type new_size = (size_type)(size.x * size.y);
613 
614  // Catch any bad_allocs thrown by vector::resize
615  try
616  {
617  // Create a new temporary array.
618  std::vector<ValueType> old_data(data_);
619 
620  // Resize the data to its new size.
621  data_.resize(new_size);
622 
623  // walk over the new grid and copy the old stuff back.
624  static ValueType default_value = (ValueType)0;
625  for (size_type i = 0; i < new_size; i++)
626  {
627  size_type x = i % size.x;
628  size_type y = i / size.x;
629  if (x >= size_.x || y >= size_.y)
630  {
631  data_[i] = default_value;
632  }
633  else
634  {
635  data_[i] = old_data[x + y * size_.x];
636  }
637  }
638 
639  // Correct the grid dimension. Origin and spacing remain constant.
640  dimension_.x *= (double)size.x / (double)size_.x;
641  dimension_.y *= (double)size.y / (double)size_.y;
642  size_ = size;
643  }
644  catch (std::bad_alloc&)
645  {
646  throw Exception::OutOfMemory(__FILE__, __LINE__, new_size * (Size)sizeof(ValueType));
647  }

812  template <typename ValueType>
813  BALL_INLINE
814  ValueType TRegularData2D<ValueType>::operator ()
815  (const typename TRegularData2D<ValueType>::CoordinateType& r) const
816  {
817  CoordinateType h(r - origin_);
818  Position x = (Position)(h.x / spacing_.x);
819  Position y = (Position)(h.y / spacing_.y);
820 
821  // correct for numerical inaccuracies
822  if (x >= (size_.x - 1))
823  {
824  x = size_.x - 2;
825  }
826  if (y >= (size_.y - 1))
827  {
828  y = size_.y - 2;
829  }
830 
831  Size l = x + size_.x * y;
832  CoordinateType r_0(origin_.x + (double)x * spacing_.x,
833  origin_.y + (double)y * spacing_.y);
834 
835  double dx = 1.0 - ((r.x - r_0.x) / spacing_.x);
836  double dy = 1.0 - ((r.y - r_0.y) / spacing_.y);
837 
838  return data_[l] * dx * dy
839  + data_[l + 1] * (1.0 - dx) * dy
840  + data_[l + size_.x] * dx * (1.0 - dy)
841  + data_[l + size_.x + 1] * (1.0 - dx) * (1.0 - dy);
842  }

993  if (!outfile.isValid())
994  throw Exception::FileNotFound(__FILE__, __LINE__, filename);
995 
996  // write all information we need to recreate the grid
997  BinaryFileAdaptor<BlockValueType> adapt_block;
998  BinaryFileAdaptor<ValueType> adapt_single;
999  BinaryFileAdaptor<float> adapt_float;
1000 
1001  BinaryFileAdaptor<Size> adapt_size;
1002 
1003  adapt_size.setData(data_.size());
1004  outfile << adapt_size;
1005 
1006  // NOTE: we do not use the binary file adaptor to write out the Vector2-variables here,
1007  // the reason is a bit stupid: the data layout of Vector2 has a three-byte "hole"
1008  // that is automatically padded by the compiler to ensure the correct alignment.
1009  // valgrind is very unhappy when we write these uninitialized values out, even
1010  // though they are entirely harmless. To prevent false-positives in the error checking,
1011  // we write out the members of the vectors instead. This even saves some bytes in the
1012  // output (not that it would matter...)
1013  adapt_float.setData(origin_.x);
1014  outfile << adapt_float;
1015  adapt_float.setData(origin_.y);
1016  outfile << adapt_float;
1017 
1018  adapt_float.setData(dimension_.x);
1019  outfile << adapt_float;
1020  adapt_float.setData(dimension_.y);
1021  outfile << adapt_float;
1022 
1023  adapt_float.setData(spacing_.x);
1024  outfile << adapt_float;
1025  adapt_float.setData(spacing_.y);
1026  outfile << adapt_float;
1027 
1028  BinaryFileAdaptor<IndexType> adapt_index;
1029  adapt_index.setData(size_);
1030  outfile << adapt_index;
