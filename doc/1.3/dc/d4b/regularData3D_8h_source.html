615  TRegularData3D<ValueType>::TRegularData3D
616  (const typename TRegularData3D<ValueType>::CoordinateType& origin,
617  const typename TRegularData3D<ValueType>::CoordinateType& x_axis,
618  const typename TRegularData3D<ValueType>::CoordinateType& y_axis,
619  const typename TRegularData3D<ValueType>::CoordinateType& z_axis,
620  const typename TRegularData3D<ValueType>::IndexType& new_size)
621  throw(Exception::OutOfMemory)
622  : data_(),
623  origin_(origin),
624  dimension_(x_axis+y_axis+z_axis),
625  spacing_(0.0),
626  size_(new_size),
627  is_orthogonal_(false)
628  {
629  // compute the spacing
630  spacing_.x = x_axis.getLength() / (new_size.x - 1.);
631  spacing_.y = y_axis.getLength() / (new_size.y - 1.);
632  spacing_.z = z_axis.getLength() / (new_size.z - 1.);
633 
634  size_type size = size_.x * size_.y * size_.z;
635  try
636  {
637  data_.resize(size);
638  }
639  catch (std::bad_alloc&)
640  {
641  data_.resize(0);
642  throw Exception::OutOfMemory(__FILE__, __LINE__, size * sizeof(ValueType));
643  }
644 
645  // prepare the mapping matrix and its inverse
646  mapping_.resize(9);
647  inverse_mapping_.resize(9);
648 
649  mapping_[0] = x_axis.x; mapping_[1] = y_axis.x; mapping_[2] = z_axis.x;
650  mapping_[3] = x_axis.y; mapping_[4] = y_axis.y; mapping_[5] = z_axis.y;
651  mapping_[6] = x_axis.z; mapping_[7] = y_axis.z; mapping_[8] = z_axis.z;
652 
653  // this is numerically instable and only works well for the "simple"
654  // cases. should be replaced by QR or an SVD
655 
656  // just for readability
657  double a = mapping_[0]; double b = mapping_[1]; double c = mapping_[2];
658  double d = mapping_[3]; double e = mapping_[4]; double f = mapping_[5];
659  double g = mapping_[6]; double h = mapping_[7]; double i = mapping_[8];
660 
661  double determinant = 1. / (a*(e*i - f*h) - b*(d*i - f*g) + c*(d*h - e*g));
662 
663  inverse_mapping_[0] = determinant * (e*i - f*h);
664  inverse_mapping_[1] = determinant * (b*i - c*h);
665  inverse_mapping_[2] = determinant * (b*f - c*e);
666 
667  inverse_mapping_[3] = determinant * (f*g - d*i);
668  inverse_mapping_[4] = determinant * (a*i - c*g);
669  inverse_mapping_[5] = determinant * (c*d - a*f);
670 
671  inverse_mapping_[6] = determinant * (d*h - e*g);
672  inverse_mapping_[7] = determinant * (b*g - a*h);
673  inverse_mapping_[8] = determinant * (a*e - b*d);
674  }

715  template <typename ValueType>
716  void TRegularData3D<ValueType>::resize(const typename TRegularData3D<ValueType>::IndexType& size)
717  throw(Exception::OutOfMemory)
718  {
719  if (is_orthogonal_)
720  {
721  // If the old size equals the new size, we're done.
722  if ((size.x == size_.x) && (size_.y == size.y) && (size_.z == size.z))
723  {
724  return;
725  }
726 
727  // If the new grid is empty, this whole thing is quite easy.
728  if ((size.x == 0) || (size.y == 0) || (size.z == 0))
729  {
730  data_.resize(0);
731  dimension_.set(0.0, 0.0, 0.0);
732  return;
733  }
734  // Compute the new array size.
735  size_type new_size = (size_type)(size.x * size.y * size.z);
736 
737  // Catch any bad_allocs thrown by vector::resize
738  try
739  {
740  // Create a new temporary array.
741  std::vector<ValueType> old_data(data_);
742 
743  // Resize the data to its new size.
744  data_.resize(new_size);
745 
746  // walk over the new grid and copy the old stuff back.
747  static ValueType default_value = (ValueType)0;
748  for (size_type i = 0; i < new_size; i++)
749  {
750  size_type x = i % size.x;
751  size_type y = (i % (size.x * size.y)) / size.x;
752  size_type z = i / (size.x * size.y);
753  if ((x >= size_.x) || (y >= size_.y) || (z >= size_.z))
754  {
755  data_[i] = default_value;
756  }
757  else
758  {
759  data_[i] = old_data[x + y * size_.x + z * size_.x * size_.y];
760  }
761  }
762 
763  // Correct the grid dimension. Origin and spacing remain constant.
764  if ((size_.x != 0) && (size_.y != 0) && (size_.z != 0))
765  {
766  dimension_.x *= (double)size.x / (double)size_.x;
767  dimension_.y *= (double)size.y / (double)size_.y;
768  dimension_.z *= (double)size.z / (double)size_.z;
769  }
770  size_ = size;
771  }
772  catch (std::bad_alloc&)
773  {
774  throw Exception::OutOfMemory(__FILE__, __LINE__, new_size * (Size)sizeof(ValueType));
775  }
