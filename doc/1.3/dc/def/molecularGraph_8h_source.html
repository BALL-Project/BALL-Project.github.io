259  template <typename Node, typename Edge>
260  bool TMolecularGraph<Node, Edge>::newEdge(const Bond& bond)
261 
262  {
263  // Create convenience aliases for atoms.
264  Atom* first = const_cast<Atom*>(bond.getFirstAtom());
265  Atom* second = const_cast<Atom*>(bond.getSecondAtom());
266 
267  // Make sure we have atoms/nodes for this bond/edge.
268  if (!atom_to_node_.has(first) || !atom_to_node_.has(second))
269  {
270  return false;
271  }
272 
273  // Make sure not to create the same edge twice.
274  if (bond_to_edge_.has(const_cast<Bond*>(&bond)))
275  {
276  return true;
277  }
278 
279  // Create the new edge und add it to the hash map relating
280  // the bond to the edge.
281  NodeItemType* first_item = atom_to_node_[first];
282  NodeItemType* second_item = atom_to_node_[second];
283  edges_.push_back(EdgeItemType(bond, first_item, second_item));
284  bond_to_edge_.insert(std::pair<Bond*, EdgeItemType*>(const_cast<Bond*>(&bond), &edges_.back()));
285  first_item->adjacent_edges_.push_back(&edges_.back());
286  second_item->adjacent_edges_.push_back(&edges_.back());
287 
288  return true;
289  }

339  {
