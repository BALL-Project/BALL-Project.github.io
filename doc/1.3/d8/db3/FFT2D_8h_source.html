598  template <typename ComplexTraits>
599  typename TFFT2D<ComplexTraits>::Complex TFFT2D<ComplexTraits>::getInterpolatedValue(const Vector2& pos) const
600  throw(Exception::OutOfGrid)
601  {
602  Complex result;
603 
604  Vector2 min = inFourierSpace_ ? minFourier_ : minPhys_;
605  Vector2 max = inFourierSpace_ ? maxFourier_ : maxPhys_;
606  double stepX = inFourierSpace_ ? stepFourierX_ : stepPhysX_;
607  double stepY = inFourierSpace_ ? stepFourierY_ : stepPhysY_;
608 
609  if ( (pos.x < min.x) || (pos.y < min.y)
610  || (pos.x > max.x) || (pos.y > max.y) )
611  {
612  throw Exception::OutOfGrid(__FILE__, __LINE__);
613  }
614 
615  Vector2 h(pos.x - min.x, pos.y - min.y);
616  double modX = fmod((double)h.x,stepX);
617  double modY = fmod((double)h.y,stepY);
618 
619  if (modX==0 && modY ==0) // we are on the grid
620  {
621  return getData(pos);
622  }
623 
624  double beforeX = floor(h.x/stepX)*stepX+ min.x;
625  double beforeY = floor(h.y/stepY)*stepY+ min.y;
626  double afterX = ceil(h.x/stepX)*stepX+ min.x;
627  double afterY = ceil(h.y/stepY)*stepY+ min.y;
628 
629  double tx = (pos.x - beforeX)/stepX;
630  double ty = (pos.y - beforeY)/stepY;
631 
632  result = getData(Vector2(beforeX,beforeY))*(typename ComplexTraits::ComplexPrecision)((1.-tx)*(1.-ty));
633  result += getData(Vector2(afterX, beforeY))*(typename ComplexTraits::ComplexPrecision)( tx *(1.-ty));
634  result += getData(Vector2(beforeX,afterY ))*(typename ComplexTraits::ComplexPrecision)((1.-tx)* ty );
635  result += getData(Vector2(afterX, afterY ))*(typename ComplexTraits::ComplexPrecision)( tx * ty );
636 
637  return result;
638  }
