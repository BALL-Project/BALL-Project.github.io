1249  Composite* findPreviousPosition(Composite* p) const
1250  {
1251  // If we are at the root of the iterator, the
1252  // decrementing it results in an invalid iterator
1253  // (past-the-end).
1254  if (p == bound_)
1255  {
1256  return 0;
1257  }
1258  // If we decrement a past-the-end-iterator, we
1259  // start at the root and "fall down" on the right
1260  // hand side following the last_child_ pointers
1261  // until we hit bottom.
1262  else if (p == 0)
1263  {
1264  if (bound_->last_child_ == 0)
1265  {
1266  return bound_;
1267  }
1268  else
1269  {
1270  p = bound_->last_child_;
1271  }
1272  while (p->last_child_ != 0)
1273  {
1274  p = p->last_child_;
1275  }
1276  }
1277  // Normally, we just grab the guy to the
1278  // left in the doubly-linked child list.
1279  else if (p->previous_ != 0)
1280  {
1281  p = p->previous_;
1282 
1283  // If the guy to the left hast children,
1284  // we do the drop on the rigth again.
1285  while (p->last_child_ != 0)
1286  {
1287  p = p->last_child_;
1288  }
1289  }
1290  // Finally, if we can't go down and we can't go
1291  // left, we have to go upwards.
1292  else if (p != bound_)
1293  {
1294  p = p->parent_;
1295  }
1296 
1297  return p;
1298  }

1300  Composite* findNextPosition(Composite* p) const
1301  {
1302  // If we are in a past-the-end position, we stay put.
1303  if (p == 0)
1304  {
1305  return 0;
1306  }
1307  // Otherwise, we try the first child. If there's one,
1308  // that's our next position.
1309  else
1310  {
1311  if (p->first_child_ != 0)
1312  {
1313  p = p->first_child_;
1314  }
1315  else
1316  {
1317  // If we are already in the root node, we are done.
1318  if (p == bound_)
1319  {
1320  return 0;
1321  }
1322  // Otherwise, we try to walk to the right at the current level.
1323  if (p->next_ != 0)
1324  {
1325  p = p->next_;
1326  }
1327  // If that doesn't work out, we'll have to climb up again.
1328  // Now, we either revisit a node we have already been to, or we
1329  // are trying to climb up *beyond* our iteration root (bound_).
1330  // In the latter case, we return a past-the-end-iterator (0).
1331  else
1332  {
1333  // If we could not walk left or right and we are at the root
1334  // again, then we are done with the iteration (this is the
1335  // case if bound_ is a leaf node).
1336  while (p->next_ == 0)
1337  {
1338  p = p->parent_;
1339  if ((p == bound_) || (p == 0))
1340  {
1341  return 0;
1342  }
1343  }
1344  p = p->next_;
1345  }
1346  }
1347  }
1348  return p;
1349  }
