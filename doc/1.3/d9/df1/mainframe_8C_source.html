218  bool Mainframe::eventFilter(QObject* sender, QEvent* event)
219  {
220  if (event->type() != QEvent::KeyPress) return false;
221 
222  QKeyEvent* e = dynamic_cast<QKeyEvent*>(event);
223 
224  if (e->key() == Qt::Key_Escape &&
225  HelpViewer::getInstance("BALLView Docu")->isWhatsThisEnabled())
226  {
227  HelpViewer::getInstance("BALLView Docu")->exitWhatsThisMode();
228  }
229 
230  QPoint point = QCursor::pos();
231  QWidget* widget = qApp->widgetAt(point);
232  if (widget == scene_ &&
233  qApp->focusWidget() != scene_)
234  {
235  scene_->keyPressEvent(e);
236  return true;
237  }
238 
239  if (e->key() == Qt::Key_Delete &&
240  RTTI::isKindOf<QTreeWidget>(*sender))
241  {
242  deleteClicked();
243  return true;
244  }
245 
246  if (e->key() == Qt::Key_Enter)
247  {
248  if (composite_manager_.getNumberOfComposites() == 0) return false;
249 
250  if (getMolecularControlSelection().size() == 0)
251  {
252  control_selection_.push_back(*composite_manager_.begin());
253  }
254 
255  MolecularStructure::getInstance(0)->centerCamera();
256  return true;
257  }
258 
259  // check all menu entries if Alt or CTRL is pressed to enable shortcuts
260  if (e->key() == Qt::Key_Alt ||
261  e->key() == Qt::Key_Control)
262  {
263  checkMenus();
264  return false;
265  }
266 
267  #ifdef BALL_PYTHON_SUPPORT
268  PyWidget::getInstance(0)->reactTo(*e);
269  e->accept();
270  #endif
271 
272  return false;
273  }
